[Unit]
Description=Mazemaker Comparison Bench Pod (Mazemaker + 5 competitor memory systems on the same bench)
After=network-online.target
Wants=network-online.target

[Pod]
PodName=bench

# All runners share a single work dir bind-mounted into the pod.
# The `:z` suffix relabels for SELinux. Host: $HOME/.bench-pod
# (the systemd %h specifier expands to the unit owner's home dir).
Volume=%h/.bench-pod:/work:z

# Loopback-only — NO published ports. Operator dashboards / matrix
# rendering happen on the host after the pod exits. If a future
# runner needs to be poked from the host during a run, it can be
# exposed by adding a PublishPort line scoped to 127.0.0.1, but
# the default posture is "no listeners visible outside the pod".

# Container references — each runner is a separate container that
# Quadlet will bring up once the corresponding .container unit lands
# in this directory in v0.2. The image tags are pinned to the same
# version the pod is shipping, so a partial upgrade can't quietly mix
# competing engine builds.
#
#   localhost/bench-mazemaker:0.1.0
#   localhost/bench-hindsight:0.1.0
#   localhost/bench-letta:0.1.0
#   localhost/bench-mem0:0.1.0
#   localhost/bench-amem:0.1.0
#   localhost/bench-cognee:0.1.0
#   localhost/bench-ollama:0.1.0
#
# v0.1 ships only the pod manifest. The Mazemaker reference runner
# runs on the host Python and writes directly into /work via the
# bind mount, which keeps the smoke-test fast for the first release.

[Install]
WantedBy=default.target
